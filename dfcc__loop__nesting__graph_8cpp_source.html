26 dfcc_loop_nesting_grapht build_loop_nesting_graph(goto_programt &goto_program)
27 {
28  natural_loops_mutablet natural_loops(goto_program);
29  dfcc_loop_nesting_grapht loop_nesting_graph;
30  // Add a node per natural loop
31  for(auto &loop : natural_loops.loop_map)
32  {
33  auto loop_head = loop.first;
34  auto &loop_instructions = loop.second;
35 
36  if(loop_instructions.size() <= 1)
37  {
38  // ignore single instruction loops of the form
39  // LOOP: goto LOOP;
40  continue;
41  }
42 
43  // Find the instruction that jumps back to `loop_head` and has the
44  // highest GOTO location number, define it as the latch.
45  auto loop_latch = loop_head;
46  for(const auto &t : loop_instructions)
47  {
48  if(
49  t->is_goto() && t->get_target() == loop_head &&
50  t->location_number > loop_latch->location_number)
51  loop_latch = t;
52  }
53 
54  DATA_INVARIANT(loop_latch != loop_head, "Natural loop latch is found");
55 
56  loop_nesting_graph.add_node(loop_head, loop_latch, loop_instructions);
57  }
58 
59  // Add edges to the graph, pointing from inner loop to outer loops.
60  // An `inner` will be considered nested in `outer` iff both its head
61  // and latch instructions are instructions of `outer`.
62  for(size_t outer = 0; outer < loop_nesting_graph.size(); ++outer)
63  {
64  const auto &outer_instructions = loop_nesting_graph[outer].instructions;
65 
66  for(size_t inner = 0; inner < loop_nesting_graph.size(); ++inner)
67  {
68  if(inner == outer)
69  continue;
70 
71  if(outer_instructions.contains(loop_nesting_graph[inner].head))
72  {
73  loop_nesting_graph.add_edge(inner, outer);
74  }
75  }
76  }
77 
78  auto topsorted_size = loop_nesting_graph.topsort().size();
79  INVARIANT(
80  topsorted_size == loop_nesting_graph.size(),
81  "loop_nesting_graph must be acyclic");
82 
83  return loop_nesting_graph;
84 }
