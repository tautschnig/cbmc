838 float sqrtf(float f)
839 {
840  __CPROVER_hide:;
841 
842  if ( f < 0.0f )
843  return 0.0f/0.0f; // NaN
844  else if (__CPROVER_isinff(f) || // +Inf only
845  f == 0.0f || // Includes -0
846  __CPROVER_isnanf(f))
847  return f;
848  else if (__CPROVER_isnormalf(f))
849  {
850  float lower=__VERIFIER_nondet_float();
851  __CPROVER_assume(lower > 0.0f);
852  __CPROVER_assume(__CPROVER_isnormalf(lower));
853  // Tighter bounds can be given but are dependent on the
854  // number of exponent and significand bits. Thus they are
855  // given implicitly...
856 
857  float lowerSquare = lower * lower;
858  __CPROVER_assume(__CPROVER_isnormalf(lowerSquare));
859 
860  float upper = nextUpf(lower);
861  float upperSquare = upper * upper; // Might be +Inf
862 
863  // Restrict these to bound f and thus compute the possible
864  // values for the square root. Note that the lower bound
865  // can be equal, this is important to catch edge cases such as
866  // 0x1.fffffep+127f and relies on the smallest normal number
867  // being a perfect square (which it will be for any sensible
868  // bit width).
869  __CPROVER_assume(lowerSquare <= f);
870  __CPROVER_assume(f < upperSquare);
871 
872  // Select between them to work out which to return
873  switch(fegetround())
874  {
875  case FE_TONEAREST :
876  return (f - lowerSquare < upperSquare - f) ? lower : upper; break;
877  case FE_UPWARD :
878  return (f - lowerSquare == 0.0f) ? lower : upper; break;
879  case FE_DOWNWARD : // Fall through
880  case FE_TOWARDZERO :
881  return (f - lowerSquare == 0.0f) ? lower : upper; break;
882  default:;
883  return __VERIFIER_nondet_float();
884  }
885 
886  }
887  else
888  {
889  //assert(fpclassify(f) == FP_SUBNORMAL);
890  //assert(f > 0.0f);
891 
892  // With respect to the algebra of floating point number
893  // all subnormals seem to be perfect squares, thus ...
894 
895  float root=__VERIFIER_nondet_float();
896  __CPROVER_assume(root >= 0.0f);
897 
898  __CPROVER_assume(root * root == f);
899 
900  return root;
901  }
902 }
